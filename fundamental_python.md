#占位符 %d 数字 %s 字符串,%f 浮点小数，前面加数字可以确定位数,又如 17.456可以写成 .3f
s1 = 72
s2 = 85
s3 = s2 - s1
k = (s3)/(s1)*100
print('增加了 %.3f %%'%k)

#如果有真实的%，可以用了两个占位符进行转转译
print('这个月增加了 %s%% 的销售额。'%(30.56))
